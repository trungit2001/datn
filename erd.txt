Table organizations {
  id varchar(50) [pk]
  organization_name varchar(100)
  address varchar(80)
  city varchar(50)
  state varchar(20)
  zip varchar(20)
  lat float
  lon float
  phone varchar(50)
  revenue float
  utilization int
}

Table patients {
  id varchar(50) [pk]
  birthdate datetime
  deathdate datetime
  ssn varchar(20)
  drivers varchar(20)
  passport varchar(20)
  prefix varchar(10)
  first varchar(50)
  last varchar(50)
  suffix varchar(10)
  maiden varchar(30)
  marital varchar(5)
  race varchar(10)
  ethnicity varchar(20)
  gender varchar(1)
  birthplace varchar(150)
  address varchar(80)
  city varchar(50)
  state varchar(20)
  county varchar(30)
  zip varchar(20)
  lat float
  lon float
  heathcare_express float
  heathcare_coverage float
}

Table payers {
  id varchar(50) [pk]
  payer_name varchar(30)
  address varchar(80)
  city varchar(50)
  state_headquartered varchar(20)
  zip varchar(20)
  phone varchar(50)
  amount_covered float
  amount_uncovered float
  revenue float
  covered_encounters int
  uncovered_encounters int
  covered_medications int
  uncovered_medications int
  covered_procedures int
  uncovered_procedures int
  covered_immunizations int
  uncovered_immunizations int
  unique_customers int
  qols_avg float
  member_months int
}

Table providers {
  id varchar(50) [pk]
  organization varchar(50) [ref: > organizations.id]
  provider_name varchar(50)
  gender varchar(1)
  speciality varchar(100)
  address varchar(80)
  city varchar(50)
  state varchar(20)
  zip varchar(20)
  lat float
  lon float
  utilization int
}

Table encounters {
  id varchar(50) [pk]
  start datetime
  stop datetime
  patient varchar(50) [ref: > patients.id]
  organization varchar(50) [ref: > organizations.id]
  provider varchar(50) [ref: > providers.id]
  payer varchar(50) [ref: > payers.id]
  encounterclass varchar(30)
  code int
  description varchar(150)
  base_encounter_cost float
  total_claim_cost float
  payer_coverage float
  reasoncode float
  reasondescription varchar(150)
}

Table allergies {
  start datetime
  stop datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(150)
}

Table careplans {
  id varchar(50) [pk]
  start datetime
  stop datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(150)
  reasoncode float
  reasondescription varchar(150)
}

Table conditions {
  start datetime
  stop datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(200)
}

Table devices {
  start datetime
  stop datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(150)
  udi varchar(250)
}

Table imaging_studies {
  id varchar(50) [pk]
  date datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  bodysite_code int
  bodysite_description varchar(150)
  modality_code varchar(10)
  modality_description varchar(100)
  sop_code varchar(50)
  sop_description varchar(150)
}

Table immunizations {
  date datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(150)
  base_cost float
}

Table medications {
  start datetime
  stop datetime
  patient varchar(50) [ref: > patients.id]
  payer varchar(50) [ref: > payers.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(300)
  base_cost float
  payer_coverage float
  dispenses int
  totalcost float
  reasoncode float
  reasondescription varchar(200)
}

Table observations {
  date datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code varchar(10)
  description varchar(300)
  value varchar (100)
  units varchar(20)
  type varchar(20)
}

Table payer_transitions {
  patient varchar(50) [ref: > patients.id]
  start_year int
  end_year int
  payer varchar(50) [ref: > payers.id]
  ownership varchar(20)
}

Table procedures {
  date datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(300)
  base_cost float
  reasoncode float
  reasondescription varchar(200)
}

Table supplies {
  date datetime
  patient varchar(50) [ref: > patients.id]
  encounter varchar(50) [ref: > encounters.id]
  code int
  description varchar(200)
  quantity int
}
